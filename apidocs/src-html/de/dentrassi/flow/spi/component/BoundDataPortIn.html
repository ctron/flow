011package de.dentrassi.flow.spi.component;
012
013import java.util.List;
014import java.util.function.Consumer;
015import java.util.function.Supplier;
016
017public class BoundDataPortIn extends AbstractSingleDataPortIn {
018
019    private final Supplier<? extends Object> initializer;
020    private final Consumer<ValueResult> consumer;
021
022    private boolean initializerLoaded;
023    private Object initializerValue;
024
025    public BoundDataPortIn(final Supplier<Object> initializer, final Consumer<ValueResult> consumer) {
026        this.initializer = initializer;
027        this.consumer = consumer;
028    }
029
030    public <T> BoundDataPortIn(final Class<T> clazz, final Supplier<T> initializer, final Supplier<T> incompatibleValue,
031            final Consumer<T> consumer) {
032        this.initializer = initializer;
033        this.consumer = value -> {
034            boolean called = false;
035
036            final List<?> values = value.getValues();
037            if (values != null) {
038                for (final Object v : value.getValues()) {
039                    if (clazz.isAssignableFrom(v.getClass())) {
040                        called = true;
041                        consumer.accept(clazz.cast(v));
042                        break;
043                    }
044                }
045            }
046            if (!called) {
047                consumer.accept(incompatibleValue.get());
048            }
049        };
050    }
051
052    @Override
053    public void update() {
054        if (this.plug != null) {
055            this.consumer.accept(this.plug.get());
056        } else {
057            if (!this.initializerLoaded) {
058                this.initializerValue = this.initializer.get();
059                this.initializerLoaded = true;
060            }
061            this.consumer.accept(ValueResult.of(this.initializerValue));
062        }
063    }
064
065}
