011package de.dentrassi.flow.spi.component;
012
013import java.lang.reflect.InvocationTargetException;
014import java.lang.reflect.Method;
015import java.util.function.Supplier;
016
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020public abstract class AnnotatedComponent extends AbstractComponent {
021
022    private static final Logger logger = LoggerFactory.getLogger(AnnotatedComponent.class);
023
024    protected AnnotatedComponent() {
025        register();
026    }
027
028    private void register() {
029        for (final Method method : getClass().getMethods()) {
030
031            final DataIn dataIn = method.getAnnotation(DataIn.class);
032            if (dataIn != null) {
033                registerDataIn(method, dataIn);
034            }
035
036            final DataOut dataOut = method.getAnnotation(DataOut.class);
037            if (dataOut != null) {
038                registerDataOut(method, dataOut);
039            }
040
041            final TriggerIn triggerIn = method.getAnnotation(TriggerIn.class);
042            if (triggerIn != null) {
043                registerTriggerIn(method, triggerIn);
044            }
045
046            final TriggerOut triggerOut = method.getAnnotation(TriggerOut.class);
047            if (triggerOut != null) {
048                registerTriggerOut(method, triggerOut);
049            }
050
051        }
052    }
053
054    private void registerDataIn(final Method method, final DataIn dataIn) {
055        if (method.getParameterCount() != 1) {
056            throw new IllegalStateException(String.format(
057                    "Method '%s' is annotated with @%s, but has %s declared parameters. It must have exactly one.",
058                    method, DataIn.class.getSimpleName(), method.getParameterCount()));
059        }
060
061        final Class<?> type = method.getParameterTypes()[0];
062
063        final String finalName;
064        {
065            String name = method.getName();
066            if (name.startsWith("set") && name.length() > 3) {
067                name = name.substring(3);
068                name = Character.toLowerCase(name.charAt(0)) + name.substring(1);
069            }
070            finalName = name;
071        }
072
073        final Supplier<Object> initializer;
074
075        if (dataIn.initialize()) {
076            initializer = () -> getInitializer(finalName, type).orElse(null);
077        } else {
078            initializer = () -> null;
079        }
080
081        registerDataIn(finalName, initializer, value -> {
082
083            Object targetValue = null;
084
085            try {
086                targetValue = value.getValueAs(type);
087            } catch (final Exception e) {
088                logger.info("Failed to convert to target value", e);
089            }
090
091            // a type mismatch will result in a null value
092
093            try {
094                method.invoke(AnnotatedComponent.this, targetValue);
095            } catch (final Exception e) {
096                throw new RuntimeException(e);
097            }
098        });
099    }
100
101    private void registerDataOut(final Method method, final DataOut dataOut) {
102        if (method.getParameterCount() != 0) {
103            throw new IllegalStateException(String.format(
104                    "Method '%s' is annotated with @%s, but has %s declared parameters. It must have none.", method,
105                    DataOut.class.getSimpleName(), method.getParameterCount()));
106        }
107
108        String name = method.getName();
109
110        if (name.startsWith("get") && name.length() > 3) {
111            name = name.substring(3);
112            name = Character.toLowerCase(name.charAt(0)) + name.substring(1);
113        } else if (name.startsWith("is") && name.length() > 2) {
114            name = name.substring(2);
115            name = Character.toLowerCase(name.charAt(0)) + name.substring(1);
116        }
117
118        final String finalName = name;
119
120        registerDataOut(finalName, () -> {
121            try {
122                return method.invoke(AnnotatedComponent.this);
123            } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
124                throw new RuntimeException("Failed to get @DataOut value through introspection", e);
125            }
126        });
127    }
128
129    private void registerTriggerIn(final Method method, final TriggerIn triggerIn) {
130        if (method.getParameterCount() != 0) {
131            throw new IllegalStateException(
132                    String.format("Method '%s' is annotated with @%s, but has declared parameters.", method,
133                            TriggerIn.class.getSimpleName()));
134        }
135
136        registerTriggerIn(method.getName(), () -> {
137            try {
138                updateAllData();
139                method.invoke(AnnotatedComponent.this);
140            } catch (final Exception e) {
141                logger.warn("Trigger In failed", e);
142                throw new RuntimeException(e);
143            }
144        });
145    }
146
147    private void registerTriggerOut(final Method method, final TriggerOut triggerOut) {
148
149        if (method.getParameterCount() != 1 || method.getParameterTypes()[0].isAssignableFrom(Runtime.class)) {
150            throw new IllegalStateException(
151                    String.format(
152                            "Method '%s' is annotated with @%s, but doesn't have exactly one argument of type %s.",
153                            method, TriggerOut.class.getSimpleName(), Runnable.class.getName()));
154        }
155
156        String name = method.getName();
157        if (name.startsWith("set") && name.length() > 3) {
158            name = name.substring(3);
159            name = Character.toLowerCase(name.charAt(0)) + name.substring(1);
160        }
161
162        final TriggerPortOut trigger = registerTriggerOut(name);
163        final Runnable runnable = trigger::execute;
164
165        try {
166            method.invoke(this, runnable);
167        } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
168            throw new RuntimeException("Failed to assing trigger out runnable", e);
169        }
170    }
171
172}
