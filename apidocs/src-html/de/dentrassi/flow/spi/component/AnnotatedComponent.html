011package de.dentrassi.flow.spi.component;
012
013import java.lang.reflect.InvocationTargetException;
014import java.lang.reflect.Method;
015import java.util.function.Supplier;
016
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020public abstract class AnnotatedComponent extends AbstractComponent {
021
022    private static final Logger logger = LoggerFactory.getLogger(AnnotatedComponent.class);
023
024    protected AnnotatedComponent() {
025        registerTriggerIn();
026        registerData();
027    }
028
029    private void registerData() {
030        for (final Method method : getClass().getMethods()) {
031
032            final DataIn dataIn = method.getAnnotation(DataIn.class);
033            if (dataIn != null) {
034                registerDataIn(method, dataIn);
035            }
036
037            final DataOut dataOut = method.getAnnotation(DataOut.class);
038            if (dataOut != null) {
039                registerDataOut(method, dataOut);
040            }
041        }
042    }
043
044    private void registerDataIn(final Method method, final DataIn dataIn) {
045        if (method.getParameterCount() != 1) {
046            throw new IllegalStateException(String.format(
047                    "Method '%s' is annotated with @%s, but has %s declared parameters. It must have exactly one.",
048                    method, DataIn.class.getSimpleName(), method.getParameterCount()));
049        }
050
051        String name = method.getName();
052        if (name.startsWith("set") && name.length() > 3) {
053            name = name.substring(3);
054            name = Character.toLowerCase(name.charAt(0)) + name.substring(1);
055        }
056
057        final String finalName = name;
058
059        final Supplier<Object> initializer;
060
061        if (dataIn.initialize()) {
062            initializer = () -> getInitializer(finalName);
063        } else {
064            initializer = () -> null;
065        }
066
067        final Class<?> type = method.getParameterTypes()[0];
068
069        registerDataIn(finalName, initializer, value -> {
070
071            Object targetValue = null;
072
073            if (value.getValues() != null) {
074                for (final Object v : value.getValues()) {
075
076                    if (v == null && value.getValues().size() == 1) {
077                        targetValue = null;
078                        break;
079                    }
080
081                    if (type.isAssignableFrom(v.getClass())) {
082                        targetValue = v;
083                        break;
084                    }
085                }
086            }
087
088            // a type mismatch will result in a null value
089
090            try {
091                method.invoke(AnnotatedComponent.this, targetValue);
092            } catch (final Exception e) {
093                throw new RuntimeException(e);
094            }
095        });
096    }
097
098    private void registerDataOut(final Method method, final DataOut dataOut) {
099        if (method.getParameterCount() != 0) {
100            throw new IllegalStateException(String.format(
101                    "Method '%s' is annotated with @%s, but has %s declared parameters. It must have none.", method,
102                    DataOut.class.getSimpleName(), method.getParameterCount()));
103        }
104
105        String name = method.getName();
106        if (name.startsWith("get") && name.length() > 3) {
107            name = name.substring(3);
108            name = Character.toLowerCase(name.charAt(0)) + name.substring(1);
109        }
110
111        final String finalName = name;
112
113        registerDataOut(finalName, () -> {
114            try {
115                return method.invoke(AnnotatedComponent.this);
116            } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
117                throw new RuntimeException("Failed to get @DataOut value through introspection", e);
118            }
119        });
120    }
121
122    private void registerTriggerIn() {
123        for (final Method method : getClass().getMethods()) {
124            final TriggerIn triggerIn = method.getAnnotation(TriggerIn.class);
125            if (triggerIn == null) {
126                continue;
127            }
128
129            if (method.getParameterCount() != 0) {
130                throw new IllegalStateException(
131                        String.format("Method '%s' is annotated with @%s, but has declared parameters.", method,
132                                TriggerIn.class.getSimpleName()));
133            }
134
135            registerTriggerIn(method.getName(), () -> {
136                try {
137                    updateAllData();
138                    method.invoke(AnnotatedComponent.this);
139                } catch (final Exception e) {
140                    logger.info("Trigger In failed", e);
141                    throw new RuntimeException(e);
142                }
143            });
144        }
145    }
146
147}
