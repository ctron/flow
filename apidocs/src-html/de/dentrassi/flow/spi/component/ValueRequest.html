011package de.dentrassi.flow.spi.component;
012
013import static java.util.Collections.singletonList;
014import static java.util.Objects.requireNonNull;
015
016import java.util.List;
017
018public final class ValueRequest {
019
020    private final List<Class<?>> types;
021
022    private ValueRequest(final Class<?> requiredType) {
023        this.types = singletonList(requiredType);
024    }
025
026    public List<Class<?>> getTypes() {
027        return this.types;
028    }
029
030    public static ValueRequest of(final Class<?> requiredType) {
031        requireNonNull(requiredType);
032        return new ValueRequest(requiredType);
033    }
034
035    /**
036     * Check if a requests contains a request for this class
037     * 
038     * @param clazz
039     *            the class to check
040     * @return {@code true } if is does
041     */
042    public boolean requests(final Class<?> clazz) {
043        for (final Class<?> requestClass : this.types) {
044            if (requestClass.isAssignableFrom(clazz)) {
045                return true;
046            }
047        }
048
049        return false;
050    }
051
052}
