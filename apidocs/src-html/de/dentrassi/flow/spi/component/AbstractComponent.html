011package de.dentrassi.flow.spi.component;
012
013import static java.util.Optional.empty;
014import static java.util.Optional.ofNullable;
015
016import java.util.HashMap;
017import java.util.Map;
018import java.util.Optional;
019import java.util.function.Consumer;
020import java.util.function.Function;
021import java.util.function.Supplier;
022
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025
026import de.dentrassi.flow.ComponentContext;
027import de.dentrassi.flow.PortType;
028import de.dentrassi.flow.spi.Component;
029import de.dentrassi.flow.spi.DataPlugIn;
030import de.dentrassi.flow.spi.DataPlugOut;
031import de.dentrassi.flow.spi.TriggerPlugIn;
032import de.dentrassi.flow.spi.TriggerPlugOut;
033
034public abstract class AbstractComponent implements Component {
035
036    private static final Logger logger = LoggerFactory.getLogger(AbstractComponent.class);
037
038    private final Map<String, TriggerPortOut> outTriggers = new HashMap<>();
039    private final Map<String, TriggerPortIn> inTriggers = new HashMap<>();
040    private final Map<String, SimpleDataPortOut> outData = new HashMap<>();
041    private final Map<String, DataPortIn> inData = new HashMap<>();
042
043    private Map<String, String> initializers;
044    private ComponentContext context;
045
046    private EventContext event;
047
048    @Override
049    public void start(final Map<String, String> initializers, final ComponentContext context,
050            final EventContext event) {
051        this.initializers = initializers;
052        this.context = context;
053        this.event = event;
054
055        this.outTriggers.keySet().forEach(name -> emitAddPort(name, PortType.TRIGGER_OUT));
056        this.inTriggers.keySet().forEach(name -> emitAddPort(name, PortType.TRIGGER_IN));
057        this.outData.keySet().forEach(name -> emitAddPort(name, PortType.DATA_OUT));
058        this.inData.keySet().forEach(name -> emitAddPort(name, PortType.DATA_IN));
059    }
060
061    @Override
062    public void stop() {
063    }
064
065    protected void runOnContext(final Runnable runnable) {
066        this.context.run(runnable);
067    }
068
069    protected <T> Optional<T> getInitializer(final String key, final Class<T> clazz) {
070
071        final String value = this.initializers.get(key);
072
073        if (value == null) {
074            return empty();
075        }
076
077        return ofNullable(this.context.geTypeConverterManager().convert(value, clazz));
078
079    }
080
081    protected void emitAddPort(final String name, final PortType type) {
082        if (this.event != null) {
083            this.event.addedPort(name, type);
084        }
085    }
086
087    protected TriggerPortOut registerTriggerOut(final String portName) {
088        final TriggerPortOut result = new TriggerPortOut();
089        this.outTriggers.put(portName, result);
090
091        emitAddPort(portName, PortType.TRIGGER_OUT);
092
093        return result;
094    }
095
096    protected void registerTriggerIn(final String portName, final Runnable runnable) {
097
098        this.inTriggers.put(portName, new TriggerPortIn(runnable));
099
100        emitAddPort(portName, PortType.TRIGGER_IN);
101    }
102
103    protected void triggerOut(final String portName) {
104        final TriggerPortOut port = this.outTriggers.get(portName);
105        port.execute();
106    }
107
108    protected void updateAllData() {
109        for (final Map.Entry<String, DataPortIn> in : this.inData.entrySet()) {
110            try {
111                in.getValue().update();
112            } catch (final Exception e) {
113                logger.warn("Failed to update port: {}", in.getKey(), e);
114                throw e;
115            }
116        }
117    }
118
119    protected DataPortOut getDataPortOut(final String portName) {
120        return this.outData.get(portName);
121    }
122
123    protected DataPortIn getDataPortIn(final String portName) {
124        return this.inData.get(portName);
125    }
126
127    protected TriggerPortOut getTriggerPortOut(final String portName) {
128        return this.outTriggers.get(portName);
129    }
130
131    protected TriggerPortIn getTriggerPortIn(final String portName) {
132        return this.inTriggers.get(portName);
133    }
134
135    protected void registerDataOut(final String portName, final Supplier<?> supplier) {
136        this.outData.put(portName, new SimpleDataPortOut(SimpleDataPortOut.singleType(supplier, () -> null)));
137        emitAddPort(portName, PortType.DATA_OUT);
138    }
139
140    protected void registerDataOut(final String portName, final Function<ValueRequest, ValueResult> supplier) {
141        this.outData.put(portName, new SimpleDataPortOut(supplier));
142        emitAddPort(portName, PortType.DATA_OUT);
143    }
144
145    protected <T extends DataPortIn> T registerDataIn(final String portName, final T port) {
146        this.inData.put(portName, port);
147        emitAddPort(portName, PortType.DATA_IN);
148        return port;
149    }
150
151    protected void registerDataIn(final String portName, final Supplier<Object> initializer,
152            final Consumer<ValueResult> consumer) {
153        registerDataIn(portName, new BoundDataPortIn(initializer, consumer));
154    }
155
156    protected <T> void registerDataIn(final String portName, final Class<T> clazz, final Supplier<T> initializer,
157            final Supplier<T> incompatibleValue, final Consumer<T> consumer) {
158        registerDataIn(portName, new BoundDataPortIn(clazz, initializer, incompatibleValue, consumer));
159    }
160
161    protected <T> void registerDataIn(final String portName, final Class<T> clazz, final Supplier<T> initializer,
162            final Consumer<T> consumer) {
163        registerDataIn(portName, new BoundDataPortIn(clazz, initializer, () -> null, consumer));
164    }
165
166    private <T> T getPort(final String portType, final String portName, final Map<String, T> ports) {
167        final T port = ports.get(portName);
168        if (port == null) {
169            throw new IllegalArgumentException(
170                    String.format("%s '%s' on component %s not found", portType, portName, this));
171        }
172        return port;
173    }
174
175    @Override
176    public void connectTriggerOut(final String portName, final TriggerPlugOut plug) {
177        getPort("Trigger Out Port", portName, this.outTriggers).add(plug);
178    }
179
180    @Override
181    public void connectTriggerIn(final String portName, final TriggerPlugIn plug) {
182        getPort("Trigger In Port", portName, this.inTriggers).add(plug);
183    }
184
185    @Override
186    public void connectDataOut(final String portName, final DataPlugOut plug) {
187        plug.setSupplier(getPort("Data Out Port", portName, this.outData));
188    }
189
190    @Override
191    public void connectDataIn(final String portName, final DataPlugIn plug) {
192        getPort("Data In Port", portName, this.inData).add(plug);
193    }
194
195}
