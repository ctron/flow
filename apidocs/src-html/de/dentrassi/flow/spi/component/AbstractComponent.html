011package de.dentrassi.flow.spi.component;
012
013import java.util.HashMap;
014import java.util.Map;
015import java.util.function.Consumer;
016import java.util.function.Function;
017import java.util.function.Supplier;
018
019import de.dentrassi.flow.Component;
020import de.dentrassi.flow.ComponentContext;
021import de.dentrassi.flow.spi.DataPlugIn;
022import de.dentrassi.flow.spi.DataPlugOut;
023import de.dentrassi.flow.spi.TriggerPlugIn;
024import de.dentrassi.flow.spi.TriggerPlugOut;
025
026public abstract class AbstractComponent implements Component {
027
028    private final Map<String, TriggerPortOut> outTriggers = new HashMap<>();
029    private final Map<String, TriggerPortIn> inTriggers = new HashMap<>();
030
031    private final Map<String, DataPortOut> outData = new HashMap<>();
032    private final Map<String, DataPortIn> inData = new HashMap<>();
033    private Map<String, String> initializers;
034
035    @Override
036    public void start(final Map<String, String> initializers, final ComponentContext context) {
037        this.initializers = initializers;
038    }
039
040    @Override
041    public void stop() {
042    }
043
044    protected String getInitializer(final String key) {
045        return this.initializers.get(key);
046    }
047
048    protected String getInitializer(final String key, final String defaultValue) {
049        return this.initializers.getOrDefault(key, defaultValue);
050    }
051
052    protected Long getInitializerLong(final String key, final Long defaultValue) {
053        final String value = this.initializers.get(key);
054
055        if (value != null) {
056            try {
057                return Long.parseLong(value);
058            } catch (final NumberFormatException e) {
059            }
060        }
061
062        return defaultValue;
063    }
064
065    protected Double getInitializerDouble(final String key, final Double defaultValue) {
066        final String value = this.initializers.get(key);
067
068        if (value != null) {
069            try {
070                return Double.parseDouble(value);
071            } catch (final NumberFormatException e) {
072            }
073        }
074
075        return defaultValue;
076    }
077
078    protected void registerTriggerOut(final String portName) {
079        this.outTriggers.put(portName, new TriggerPortOut());
080    }
081
082    protected void registerTriggerIn(final String portName, final Runnable runnable) {
083        this.inTriggers.put(portName, new TriggerPortIn(runnable));
084    }
085
086    protected void triggerOut(final String portName) {
087        final TriggerPortOut port = this.outTriggers.get(portName);
088        port.execute();
089    }
090
091    protected void updateAllData() {
092        for (final DataPortIn in : this.inData.values()) {
093            in.update();
094        }
095    }
096
097    protected DataPortOut getDataPortOut(final String portName) {
098        return this.outData.get(portName);
099    }
100
101    protected DataPortIn getDataPortIn(final String portName) {
102        return this.inData.get(portName);
103    }
104
105    protected TriggerPortOut getTriggerPortOut(final String portName) {
106        return this.outTriggers.get(portName);
107    }
108
109    protected TriggerPortIn getTriggerPortIn(final String portName) {
110        return this.inTriggers.get(portName);
111    }
112
113    protected void registerDataOut(final String portName, final Supplier<?> supplier) {
114        this.outData.put(portName, new DataPortOut(DataPortOut.singleType(supplier, () -> null)));
115    }
116
117    protected void registerDataOut(final String portName, final Function<ValueRequest, ValueResult> supplier) {
118        this.outData.put(portName, new DataPortOut(supplier));
119    }
120
121    protected <T extends DataPortIn> T registerDataIn(final String portName, final T port) {
122        this.inData.put(portName, port);
123        return port;
124    }
125
126    protected void registerDataIn(final String portName, final Supplier<Object> initializer,
127            final Consumer<ValueResult> consumer) {
128        registerDataIn(portName, new BoundDataPortIn(initializer, consumer));
129    }
130
131    protected <T> void registerDataIn(final String portName, final Class<T> clazz, final Supplier<T> initializer,
132            final Supplier<T> incompatibleValue, final Consumer<T> consumer) {
133        registerDataIn(portName, new BoundDataPortIn(clazz, initializer, incompatibleValue, consumer));
134    }
135
136    protected <T> void registerDataIn(final String portName, final Class<T> clazz, final Supplier<T> initializer,
137            final Consumer<T> consumer) {
138        registerDataIn(portName, new BoundDataPortIn(clazz, initializer, () -> null, consumer));
139    }
140
141    private <T> T getPort(final String portType, final String portName, final Map<String, T> ports) {
142        final T port = ports.get(portName);
143        if (port == null) {
144            throw new IllegalArgumentException(
145                    String.format("%s '%s' on component %s not found", portType, portName, this));
146        }
147        return port;
148    }
149
150    @Override
151    public void connectTriggerOut(final String portName, final TriggerPlugOut plug) {
152        getPort("Trigger Out Port", portName, this.outTriggers).add(plug);
153    }
154
155    @Override
156    public void connectTriggerIn(final String portName, final TriggerPlugIn plug) {
157        getPort("Trigger In Port", portName, this.inTriggers).add(plug);
158    }
159
160    @Override
161    public void connectDataOut(final String portName, final DataPlugOut plug) {
162        plug.setSupplier(getPort("Data Out Port", portName, this.outData));
163    }
164
165    @Override
166    public void connectDataIn(final String portName, final DataPlugIn plug) {
167        getPort("Data In Port", portName, this.inData).add(plug);
168    }
169
170}
