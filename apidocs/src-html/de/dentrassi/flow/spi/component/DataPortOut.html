011package de.dentrassi.flow.spi.component;
012
013import static java.util.Objects.requireNonNull;
014
015import java.util.function.Function;
016import java.util.function.Supplier;
017
018public class DataPortOut {
019
020    private final Function<ValueRequest, ValueResult> supplier;
021
022    public DataPortOut(final Function<ValueRequest, ValueResult> supplier) {
023        this.supplier = supplier;
024    }
025
026    public ValueResult get(final ValueRequest request) {
027        return this.supplier.apply(request);
028    }
029
030    public static Function<ValueRequest, ValueResult> singleType(final Supplier<?> supplier,
031            final Supplier<?> defaultValue) {
032
033        requireNonNull(supplier);
034        requireNonNull(defaultValue);
035
036        return request -> {
037            final Object value = supplier.get();
038            if (value == null) {
039                return null;
040            }
041
042            final Class<?> valueClass = value.getClass();
043            for (final Class<?> type : request.getTypes()) {
044                if (type.isAssignableFrom(valueClass)) {
045                    return ValueResult.of(value);
046                }
047            }
048
049            return ValueResult.of(defaultValue.get());
050        };
051    }
052}
