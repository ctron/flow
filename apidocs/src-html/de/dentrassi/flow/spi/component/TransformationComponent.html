011package de.dentrassi.flow.spi.component;
012
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015
016public abstract class TransformationComponent extends AbstractComponent {
017
018    private static final Logger logger = LoggerFactory.getLogger(TransformationComponent.class);
019
020    private final SimpleSingleDataPortIn input;
021
022    protected TransformationComponent() {
023        this.input = registerDataIn("input", new SimpleSingleDataPortIn());
024        registerDataOut("output", this::process);
025    }
026
027    private ValueResult process(final ValueRequest request) {
028
029        // update source value
030
031        this.input.update();
032
033        // process error
034
035        final ValueResult value = this.input.getValue();
036        if (value.isError()) {
037            logger.debug("Source is errored");
038            return value;
039        }
040
041        // trigger conversion
042
043        try {
044            final ValueResult result = convert(request, value);
045            logger.debug("Transformed[{}] - {} / {} = {}", getClass(), request, value, result);
046            return result;
047        } catch (final Exception e) {
048            logger.info("Failed to transform value", e);
049            return ValueResult.ofError(e);
050        }
051    }
052
053    public abstract ValueResult convert(ValueRequest request, ValueResult input) throws Exception;
054
055}
