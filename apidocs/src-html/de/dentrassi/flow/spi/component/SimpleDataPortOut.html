011package de.dentrassi.flow.spi.component;
012
013import static java.util.Objects.requireNonNull;
014
015import java.util.function.Function;
016import java.util.function.Supplier;
017
018public class SimpleDataPortOut implements DataPortOut {
019
020    private final Function<ValueRequest, ValueResult> supplier;
021
022    public SimpleDataPortOut(final Function<ValueRequest, ValueResult> supplier) {
023        requireNonNull(supplier);
024        this.supplier = supplier;
025    }
026
027    /* (non-Javadoc)
028     * @see de.dentrassi.flow.spi.component.DataPortOut#get(de.dentrassi.flow.spi.component.ValueRequest)
029     */
030    @Override
031    public ValueResult get(final ValueRequest request) {
032        return requireNonNull(this.supplier.apply(request));
033    }
034
035    public static Function<ValueRequest, ValueResult> singleType(final Supplier<?> supplier) {
036        return singleType(supplier, () -> ValueResult.ofError("No compatible type found"));
037    }
038
039    public static Function<ValueRequest, ValueResult> singleType(final Supplier<?> supplier,
040            final Supplier<ValueResult> defaultValue) {
041
042        requireNonNull(supplier);
043        requireNonNull(defaultValue);
044
045        return request -> {
046            final Object value = supplier.get();
047
048            if (value == null) {
049                return null;
050            }
051
052            final Class<?> valueClass = value.getClass();
053
054            for (final Class<?> type : request.getTypes()) {
055                if (type.isAssignableFrom(valueClass)) {
056                    return ValueResult.of(value);
057                }
058            }
059
060            return defaultValue.get();
061        };
062    }
063}
