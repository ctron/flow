011package de.dentrassi.flow.model;
012
013import java.util.Optional;
014
015import de.dentrassi.flow.PortType;
016import de.dentrassi.flow.event.AddComponent;
017import de.dentrassi.flow.event.AddConnection;
018import de.dentrassi.flow.event.AddPort;
019import de.dentrassi.flow.event.FlowEvent;
020import de.dentrassi.flow.event.FlowListener;
021
022public class ModelListener implements FlowListener {
023
024    private final Flow flow;
025
026    public ModelListener() {
027        this.flow = FlowFactory.eINSTANCE.createFlow();
028    }
029
030    @Override
031    public void flowEvent(final FlowEvent event) {
032        if (event instanceof AddComponent) {
033            addComponent((AddComponent) event);
034        } else if (event instanceof AddPort) {
035            addPort((AddPort) event);
036        } else if (event instanceof AddConnection) {
037            addConnection((AddConnection) event);
038        }
039    }
040
041    private void addPort(final AddPort event) {
042        final Optional<Node> component = component(event.getPort().getComponent().getId());
043
044        component.ifPresent(n -> {
045            final Port port = createPort(event.getPort(), event.getType());
046            if (port != null) {
047                n.getPorts().add(port);
048            }
049        });
050    }
051
052    private Optional<Node> component(final String id) {
053        final Optional<Node> node = this.flow
054                .getNodes()
055                .stream()
056                .filter(n -> n.getId().equals(id)).findFirst();
057        return node;
058    }
059
060    private Port createPort(final de.dentrassi.flow.Port port, final PortType type) {
061
062        switch (type) {
063        case DATA_IN: {
064            final Port result = FlowFactory.eINSTANCE.createDataInPort();
065            result.setName(port.getPortName());
066            return result;
067        }
068        case DATA_OUT: {
069            final Port result = FlowFactory.eINSTANCE.createDataOutPort();
070            result.setName(port.getPortName());
071            return result;
072        }
073        case TRIGGER_IN: {
074            final Port result = FlowFactory.eINSTANCE.createTriggerInPort();
075            result.setName(port.getPortName());
076            return result;
077        }
078        case TRIGGER_OUT: {
079            final Port result = FlowFactory.eINSTANCE.createTriggerOutPort();
080            result.setName(port.getPortName());
081            return result;
082        }
083        default:
084            return null;
085        }
086    }
087
088    private void addComponent(final AddComponent event) {
089        final Node node = FlowFactory.eINSTANCE.createNode();
090        node.setId(event.getId());
091        node.setType(event.getType());
092        this.flow.getNodes().add(node);
093    }
094
095    private void addConnection(final AddConnection event) {
096
097        switch (event.getConnectionType()) {
098        case DATA: {
099            final DataConnection connection = FlowFactory.eINSTANCE.createDataConnection();
100
101            dataPortOut(event.getOut()).ifPresent(out -> {
102                dataPortIn(event.getIn()).ifPresent(in -> {
103                    connection.setOut(out);
104                    connection.setIn(in);
105                    this.flow.getConnections().add(connection);
106                });
107            });
108        }
109        case TRIGGER: {
110            final TriggerConnection connection = FlowFactory.eINSTANCE.createTriggerConnection();
111
112            triggerPortOut(event.getOut()).ifPresent(out -> {
113                triggerPortIn(event.getIn()).ifPresent(in -> {
114                    connection.setOut(out);
115                    connection.setIn(in);
116                    this.flow.getConnections().add(connection);
117                });
118            });
119        }
120        default:
121            return;
122        }
123
124    }
125
126    private Optional<DataInPort> dataPortIn(final de.dentrassi.flow.Port in) {
127        final Optional<Node> component = component(in.getComponent().getId());
128        return component.flatMap(c -> {
129            return c.getPorts().stream().filter(p -> p instanceof DataInPort && p.getName().equals(in.getPortName()))
130                    .map(p -> (DataInPort) p)
131                    .findAny();
132        });
133    }
134
135    private Optional<DataOutPort> dataPortOut(final de.dentrassi.flow.Port in) {
136        final Optional<Node> component = component(in.getComponent().getId());
137        return component.flatMap(c -> {
138            return c.getPorts().stream().filter(p -> p instanceof DataOutPort && p.getName().equals(in.getPortName()))
139                    .map(p -> (DataOutPort) p)
140                    .findAny();
141        });
142    }
143
144    private Optional<TriggerInPort> triggerPortIn(final de.dentrassi.flow.Port in) {
145        final Optional<Node> component = component(in.getComponent().getId());
146        return component.flatMap(c -> {
147            return c.getPorts().stream().filter(p -> p instanceof TriggerInPort && p.getName().equals(in.getPortName()))
148                    .map(p -> (TriggerInPort) p)
149                    .findAny();
150        });
151    }
152
153    private Optional<TriggerOutPort> triggerPortOut(final de.dentrassi.flow.Port in) {
154        final Optional<Node> component = component(in.getComponent().getId());
155        return component.flatMap(c -> {
156            return c.getPorts().stream()
157                    .filter(p -> p instanceof TriggerOutPort && p.getName().equals(in.getPortName()))
158                    .map(p -> (TriggerOutPort) p)
159                    .findAny();
160        });
161    }
162
163    public Flow getFlow() {
164        return this.flow;
165    }
166
167}
