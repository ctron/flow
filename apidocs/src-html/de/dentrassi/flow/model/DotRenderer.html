011package de.dentrassi.flow.model;
012
013import java.io.PrintWriter;
014import java.io.Writer;
015import java.util.Arrays;
016import java.util.HashMap;
017import java.util.Map;
018import java.util.Set;
019import java.util.TreeSet;
020import java.util.stream.Collectors;
021
022public class DotRenderer implements Renderer {
023
024    private DotRenderer() {
025    }
026
027    @Override
028    public void render(final Flow flow, final Writer writer) {
029
030        final PrintWriter out = new PrintWriter(writer);
031
032        out.println("digraph flow {");
033
034        out.println("graph [rankdir=\"LR\"];");
035        out.println("node [shape=record];");
036        out.println("edge [minlen=2];");
037
038        for (final Node n : flow.getNodes()) {
039
040            final Set<String> ins = new TreeSet<>();
041            final Set<String> outs = new TreeSet<>();
042
043            for (final Port p : n.getPorts()) {
044
045                if (p instanceof DataInPort || p instanceof TriggerInPort) {
046                    ins.add(String.format("<%1$s> %1$s", p.getName()));
047                } else if (p instanceof DataOutPort || p instanceof TriggerOutPort) {
048                    outs.add(String.format("<%1$s> %1$s", p.getName()));
049                }
050
051            }
052
053            final String inGroup = "{" + String.join("|", ins) + "}";
054            final String outGroup = "{" + String.join("|", outs) + "}";
055
056            String type = n.getType();
057            final String toks[] = type.split("\\.");
058            if (toks.length > 1) {
059                final String sn = toks[toks.length - 1];
060                type = sn + "\n"
061                        + Arrays.stream(toks, 0, toks.length - 1).collect(Collectors.joining("."));
062            }
063
064            final String label = String.format("{ %s | %s | %s }", inGroup, type, outGroup);
065
066            out.format("\"%s\"[shape=record,label=\"%s\"]%n", n.getId(),
067                    label);
068        }
069
070        out.println();
071
072        for (final Connection c : flow.getConnections()) {
073            if (c instanceof DataConnection) {
074
075                final DataOutPort outPort = ((DataConnection) c).getOut();
076                final DataInPort inPort = ((DataConnection) c).getIn();
077
078                final Map<String, String> attributes = new HashMap<>();
079                attributes.put("arrowhead", "none");
080                attributes.put("arrowtail", "inv");
081                attributes.put("dir", "both");
082                attributes.put("color", "blue");
083
084                // edge(out, outPort.getNode().getId(), inPort.getNode().getId(), attributes);
085                edge(out, outPort, inPort, attributes);
086
087            } else if (c instanceof TriggerConnection) {
088
089                final TriggerOutPort outPort = ((TriggerConnection) c).getOut();
090                final TriggerInPort inPort = ((TriggerConnection) c).getIn();
091
092                final Map<String, String> attributes = new HashMap<>();
093                attributes.put("color", "red");
094
095                edge(out, outPort, inPort, attributes);
096
097            }
098        }
099
100        out.println("}");
101
102    }
103
104    public static DotRenderer newRenderer() {
105        return new DotRenderer();
106    }
107
108    private static void edge(final PrintWriter writer, final Port out, final Port in,
109            final Map<String, String> attributes) {
110
111        edge(writer,
112                String.format("\"%s\":\"%s\":e", out.getNode().getId(), out.getName()),
113                String.format("\"%s\":\"%s\":w", in.getNode().getId(), in.getName()),
114                attributes);
115
116    }
117
118    private static void edge(final PrintWriter writer, final String out, final String in,
119            final Map<String, String> attributes) {
120
121        writer.append(out);
122
123        writer.append(" -> ");
124
125        writer.append(in);
126
127        final String r = attributes.entrySet().stream().map(entry -> {
128
129            final String key = entry.getKey();
130            final String value = entry.getValue();
131
132            return key + "=\"" + value + "\"";
133        }).collect(Collectors.joining(",", "[", "]"));
134
135        writer.append(r);
136
137        writer.println();
138
139    }
140
141}
