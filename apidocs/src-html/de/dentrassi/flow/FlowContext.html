011package de.dentrassi.flow;
012
013import java.util.Map;
014
015public interface FlowContext extends ComponentInstance {
016
017    public default ComponentInstance createComponent(final String componentType,
018            final Map<String, String> initializers) {
019        return createComponent(null, componentType, initializers);
020    }
021
022    public default ComponentInstance createComponent(final String componentType) {
023        return createComponent(null, componentType, null);
024    }
025
026    public ComponentInstance createComponent(String id, String componentType, Map<String, String> initializers);
027
028    public Connection connectTrigger(Port out, Port in);
029
030    public default Connection connectData(final Port out, final Port in) {
031        return connectData(out, in, Object.class);
032    }
033
034    public Connection connectData(Port out, Port in, Class<?> requiredType);
035
036    public default Port triggerOutInit() {
037        return Port.port(this, "init");
038    }
039}
