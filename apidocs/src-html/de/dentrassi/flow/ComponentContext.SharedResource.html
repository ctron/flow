011package de.dentrassi.flow;
012
013import java.util.function.Consumer;
014import java.util.function.Supplier;
015
016import de.dentrassi.flow.type.TypeConverterManager;
017
018public interface ComponentContext {
019
020    public interface SharedResource<T> extends AutoCloseable {
021
022        public T get();
023
024        @Override
025        public void close();
026    }
027
028    public <T> SharedResource<T> createSharedResource(String namespace, String key, Class<T> clazz,
029            Supplier<T> supplier, Consumer<T> closeHandler);
030
031    public default <T> SharedResource<T> createSharedResource(final Class<?> namespace, final String key,
032            final Class<T> clazz, final Supplier<T> supplier, final Consumer<T> closeHandler) {
033        return createSharedResource(namespace.getName(), key, clazz, supplier, closeHandler);
034    }
035
036    /**
037     * Schedule to run on the context thread.
038     * 
039     * @param runnable
040     *            the runnable to run
041     */
042    public void run(Runnable runnable);
043
044    public TypeConverterManager geTypeConverterManager();
045}
