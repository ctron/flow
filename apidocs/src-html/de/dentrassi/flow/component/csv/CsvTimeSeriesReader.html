068    public CsvTimeSeriesReader() {
069        registerTriggerIn("open", this::open);
070        registerTriggerIn("close", this::close);
071
072        registerDataIn("file", String.class,
073                () -> getInitializer("file", String.class).orElse(null),
074                () -> null,
075                this::setFile);
076
077        registerDataIn("timestampColumn", String.class,
078                () -> getInitializer("timestampColumn", String.class).orElse(null),
079                () -> null,
080                this::setTimestampColumn);
081
082        // TODO: Make this a mixed type port (String, TimeUnit)
083        registerDataIn("timestampUnit", String.class,
084                () -> getInitializer("timestampUnit", String.class).orElse("MS"),
085                () -> null,
086                this::setTimestampUnit);
087
088        registerDataIn("durationMultipliedBy", Long.class,
089                () -> getInitializer("durationMultipliedBy", Long.class).orElse(null),
090                () -> null, this::setDurationMultipliedBy);
091        registerDataIn("durationDividedBy", Long.class,
092                () -> getInitializer("durationDividedBy", Long.class).orElse(null), () -> null,
093                this::setDurationDividedBy);
094
095        registerDataOut("record", this::getRecord);
096
097        registerTriggerOut("updated");
098        registerTriggerOut("completed");
099    }

125        this.timestampColumn = timestampColumn;
126    }
127

129        this.durationMultipliedBy = durationMultipliedBy;
130    }
131
132    private void setDurationDividedBy(final Long durationDividedBy) {
133        this.durationDividedBy = durationDividedBy;
134    }
135
136    private void setTimestampUnit(final String timestampUnit) {
137        if (timestampUnit == null || timestampUnit.isEmpty()) {
138            this.timestampUnit = TimeUnit.SECONDS;
139        } else {
140            switch (timestampUnit.toUpperCase()) {
141
142            case "MICRO": //$FALL-THROUGH$
143            case "MICROS":
144                this.timestampUnit = TimeUnit.MICROSECONDS;
145                break;
146
147            case "MS": //$FALL-THROUGH$
148            case "MILLIS":
149                this.timestampUnit = TimeUnit.MILLISECONDS;
150                break;
151
152            case "S": //$FALL-THROUGH$
153            case "SEC":
154                this.timestampUnit = TimeUnit.SECONDS;
155                break;
156
157            case "MIN":
158                this.timestampUnit = TimeUnit.MINUTES;
159                break;
160
161            default:
162                this.timestampUnit = TimeUnit.valueOf(timestampUnit.toUpperCase());
163                break;
164            }
165        }
166
167    }
168
169    public void open() {
170
171        logger.info("Opening CSV");
172
173        updateAllData();
174
175        if (this.file == null) {
176            logger.debug("No file configured");
177            return;
178        }
179
180        final Path path = Paths.get(this.file);
181        if (!Files.isReadable(path)) {
182            logger.info("File {} is not readable", this.file);
183            return;
184        }
185
186        try {
187            this.parser = CSVParser.parse(path, StandardCharsets.UTF_8, CSVFormat.DEFAULT.withFirstRecordAsHeader());
188            this.iterator = this.parser.iterator();
189
190            if (this.iterator.hasNext()) {
191                this.nextRecord = this.iterator.next();
192                readNext();
193            } else {
194                fireEndOfFile();
195            }
196
197        } catch (final IOException e) {
198            logger.info("Failed to open reader", e);
199        }
200    }
201
202    public void close() {
203        this.iterator = null;
204        if (this.parser != null) {
205            try {
206                this.parser.close();
207            } catch (final IOException e) {
208                logger.info("Failed to close reader", e);
209            }
210            this.parser = null;
211        }
212    }
213
214    protected void readNext() {
215        if (this.parser == null) {
216            return;
217        }
218
219        setCurrentRecord(this.nextRecord);
220        fireUpdated();
221
222        if (!this.iterator.hasNext()) {
223            this.nextRecord = null;
224            logger.info("Reached end of input");
225            fireEndOfFile();
226            return;
227        }
228
229        this.nextRecord = this.iterator.next();
230
231        Duration duration;
232        try {
233            final Instant now = Instant.ofEpochMilli(fromRecord(this.record));
234            final Instant next = Instant.ofEpochMilli(fromRecord(this.nextRecord));
235            duration = Duration.between(now, next);
236        } catch (final Exception e) {
237            logger.info("Failed to parse timstamps", e);
238            duration = Duration.ofMillis(1);
239        }
240
241        triggerNextRead(duration);
242    }
243
244    private void triggerNextRead(final Duration originalDuration) {
245
246        Duration duration = originalDuration;
247
248        if (duration.isNegative()) {
249
250            duration = Duration.ZERO;
251            logger.debug("Zeroed negative duration");
252
253        } else {
254
255            duration = originalDuration;
256            if (this.durationMultipliedBy != null) {
257                duration = duration.multipliedBy(this.durationMultipliedBy);
258            }
259            if (this.durationDividedBy != null) {
260                duration = duration.dividedBy(this.durationDividedBy);
261            }
262            logger.debug("Applied transformations: {} -> {}", originalDuration, duration);
263
264        }
265
266        logger.debug("Schedule next read in: {}", duration);
267
268        final Instant now = Instant.now();
269
270        if (this.nextUpdate != null) {
271            final Duration diff = Duration.between(this.nextUpdate, now);
272            final Duration corrected = duration.minus(diff);
273            logger.debug("Correcting duration - {} -> {} (diff: {})", duration, corrected, diff);
274            duration = corrected;
275        }
276
277        this.nextUpdate = now.plus(duration);
278
279        Durations.consume(duration,
280                (delay, unit) -> this.executor.get().schedule(() -> this.context.run(this::readNext), delay, unit));
281    }
282
283    private void setCurrentRecord(final CSVRecord record) {
284        logger.debug("Current record: {}", record);
285        this.record = record;
286    }
287
288    private long fromRecord(final CSVRecord record) throws Exception {
289        try {
290            final long ts = Long.parseLong(record.get(this.timestampColumn));
291            logger.trace("Raw TS value: {}", ts);
292            return TimeUnit.MILLISECONDS.convert(ts, this.timestampUnit);
293        } catch (final Exception e) {
294            throw new Exception("Failed to parse timestamp", e);
295        }
296    }
297
298    private void fireUpdated() {
299        triggerOut("updated");
300    }
301
302    private void fireEndOfFile() {
303        triggerOut("completed");
304    }
305
306    @Override
307    public void connectDataOut(final String portName, final DataPlugOut plug) {
308        if (portName.startsWith("record/")) {
309            if (getDataPortOut(portName) == null) {
310                final String name = portName.substring("record/".length());
311                registerDataOut(portName, () -> getRecord(name));
312            }
313        }
314
315        super.connectDataOut(portName, plug);
316    }
317
318    private ValueResult getRecord(final ValueRequest request) {
319        return ValueResult.of(this.record, this.record.toMap());
320    }
321
322    private String getRecord(final String columnName) {
323        if (this.record == null) {
324            return null;
325        }
326
327        return this.record.get(columnName);
328    }
