011package de.dentrassi.flow.component.kapua;
012
013import org.eclipse.kapua.client.gateway.Client;
014import org.eclipse.kapua.client.gateway.Credentials;
015import org.eclipse.kapua.client.gateway.Transport.ListenerHandle;
016import org.eclipse.kapua.client.gateway.mqtt.fuse.FuseChannel;
017import org.eclipse.kapua.client.gateway.mqtt.fuse.FuseChannel.Builder;
018import org.eclipse.kapua.client.gateway.profile.kura.KuraMqttProfile;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022import de.dentrassi.flow.spi.component.AnnotatedComponent;
023import de.dentrassi.flow.spi.component.DataIn;
024import de.dentrassi.flow.spi.component.DataOut;
025import de.dentrassi.flow.spi.component.TriggerIn;
026import de.dentrassi.flow.spi.component.TriggerOut;
027
028public class KapuaClient extends AnnotatedComponent {
029
030    private static final Logger logger = LoggerFactory.getLogger(KapuaClient.class);
031
032    private String accountName;
033    private String clientId;
034    private String brokerUrl;
035    private String username;
036    private String password;
037    private Client client;
038    private ListenerHandle listenerHandle;
039    private boolean connected;
040    private Runnable connectionEstablished;
041    private Runnable connectionLost;
042    private Runnable clientCreated;
043    private Runnable clientDisposed;
044
045    @DataIn
046    public void setAccountName(final String accountName) {
047        this.accountName = accountName;
048    }
049
050    @DataIn
051    public void setClientId(final String clientId) {
052        this.clientId = clientId;
053    }
054
055    @DataIn
056    public void setBrokerUrl(final String brokerUrl) {
057        this.brokerUrl = brokerUrl;
058    }
059
060    @DataIn
061    public void setUsername(final String username) {
062        this.username = username;
063    }
064
065    @DataIn
066    public void setPassword(final String password) {
067        this.password = password;
068    }
069
070    @DataOut
071    public Client getClient() {
072        return this.client;
073    }
074
075    @TriggerOut
076    public void setConnectionEstablished(final Runnable connectionEstablished) {
077        this.connectionEstablished = connectionEstablished;
078    }
079
080    @TriggerOut
081    public void setConnectionLost(final Runnable connectionLost) {
082        this.connectionLost = connectionLost;
083    }
084
085    @TriggerOut
086    public void setClientCreated(final Runnable clientCreated) {
087        this.clientCreated = clientCreated;
088    }
089
090    @TriggerOut
091    public void setClientDisposed(final Runnable clientDisposed) {
092        this.clientDisposed = clientDisposed;
093    }
094
095    @DataOut
096    public boolean isConnected() {
097        return this.connected;
098    }
099
100    @TriggerIn
101    public void connect() throws Exception {
102
103        logger.info("Connecting client - broker: {}, clientId: {}", this.brokerUrl, this.clientId);
104
105        final KuraMqttProfile<Builder> builder = KuraMqttProfile.newProfile(FuseChannel.Builder::new)
106                .accountName(this.accountName)
107                .clientId(this.clientId)
108                .brokerUrl(this.brokerUrl);
109
110        if (this.username != null && this.password != null) {
111            builder.credentials(Credentials.userAndPassword(this.username, this.password));
112        }
113
114        this.client = builder.build();
115        this.listenerHandle = this.client.transport().listen(connected -> {
116            runOnContext(() -> {
117                handleConnectionStateChange(connected);
118            });
119        });
120
121        logger.info("Created client - broker: {}, clientId: {}", this.brokerUrl, this.clientId);
122
123        this.clientCreated.run();
124    }
125
126    @TriggerIn
127    public void disconnect() throws Exception {
128        this.clientDisposed.run();
129
130        if (this.listenerHandle != null) {
131            this.listenerHandle.close();
132            this.listenerHandle = null;
133        }
134        if (this.client != null) {
135            this.client.close();
136            this.client = null;
137        }
138    }
139
140    private void handleConnectionStateChange(final boolean connected) {
141        logger.info("Connection state changed: {}", connected);
142
143        this.connected = connected;
144        if (connected) {
145            this.connectionEstablished.run();
146        } else {
147            this.connectionLost.run();
148        }
149    }
150
151}
