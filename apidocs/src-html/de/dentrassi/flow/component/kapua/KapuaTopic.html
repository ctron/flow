011package de.dentrassi.flow.component.kapua;
012
013import java.util.Map;
014import java.util.Map.Entry;
015import java.util.stream.Collectors;
016
017import org.eclipse.kapua.client.gateway.Application;
018import org.eclipse.kapua.client.gateway.Payload;
019import org.eclipse.kapua.client.gateway.Topic;
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022
023import de.dentrassi.flow.spi.component.AnnotatedComponent;
024import de.dentrassi.flow.spi.component.DataIn;
025import de.dentrassi.flow.spi.component.TriggerIn;
026
027public class KapuaTopic extends AnnotatedComponent {
028
029    private static final Logger logger = LoggerFactory.getLogger(KapuaTopic.class);
030
031    private Application application;
032
033    private Topic topic;
034
035    private Map<?, ?> payload;
036
037    @DataIn(initialize = false)
038    public void setApplication(final Application application) {
039        this.application = application;
040    }
041
042    @DataIn
043    public void setTopic(final String topic) {
044        this.topic = Topic.split(topic);
045    }
046
047    @DataIn
048    public void setPayload(final Map<?, ?> payload) {
049        this.payload = payload;
050    }
051
052    @TriggerIn
053    public void publish() {
054        if (this.payload == null) {
055            return;
056        }
057
058        final Map<String, ?> payload = this.payload.entrySet()
059                .stream()
060                .collect(Collectors.toMap(e -> e.getKey().toString(), Entry::getValue));
061
062        this.application.data(this.topic).send(Payload.of(payload))
063                .whenComplete((v, e) -> {
064                    if (e != null) {
065                        logger.warn("Failed to publish", e);
066                    }
067                });
068    }
069}
