011package de.dentrassi.flow.component.mqtt;
012
013import java.util.Map;
014
015import de.dentrassi.flow.ComponentContext;
016import de.dentrassi.flow.ComponentContext.SharedResource;
017import de.dentrassi.flow.spi.component.AnnotatedComponent;
018import de.dentrassi.flow.spi.component.DataIn;
019import de.dentrassi.flow.spi.component.DataOut;
020import de.dentrassi.flow.spi.component.TriggerIn;
021import io.vertx.core.Vertx;
022import io.vertx.mqtt.MqttClientOptions;
023
024public class MqttClient extends AnnotatedComponent {
025
026    private String host;
027    private int port;
028    private String clientId;
029    private String username;
030    private String password;
031
032    private SharedResource<Vertx> vertx;
033    private io.vertx.mqtt.MqttClient client;
034
035    public MqttClient() {
036        super();
037
038        registerTriggerOut("connected");
039    }
040
041    @Override
042    public void start(final Map<String, String> initializers, final ComponentContext context) {
043        super.start(initializers, context);
044        this.vertx = context.createSharedResource(MqttClient.class, "vertx", Vertx.class, () -> Vertx.vertx(),
045                Vertx::close);
046    }
047
048    @Override
049    public void stop() {
050        disconnect();
051        this.vertx.close();
052        super.stop();
053    }
054
055    @TriggerIn
056    public void connect() {
057        final MqttClientOptions options = new MqttClientOptions();
058
059        if (this.clientId != null) {
060            options.setAutoGeneratedClientId(false);
061            options.setClientId(this.clientId);
062        } else {
063            options.setAutoGeneratedClientId(true);
064        }
065
066        options.setAutoKeepAlive(true);
067        options.setUsername(this.username);
068        options.setPassword(this.password);
069
070        this.client = io.vertx.mqtt.MqttClient.create(this.vertx.get(), options);
071        this.client.connect(this.port, this.host, connected -> {
072            connected();
073        });
074    }
075
076    @TriggerIn
077    public void disconnect() {
078        this.client.disconnect();
079    }
080
081    private void connected() {
082        triggerOut("connected");
083    }
084
085    @DataIn
086    public void setHost(final String host) {
087        this.host = host;
088    }
089
090    @DataIn
091    public void setPort(final String port) {
092        this.port = port == null ? 1883 : Integer.parseInt(port);
093    }
094
095    @DataIn
096    public void setClientId(final String clientId) {
097        this.clientId = clientId;
098    }
099
100    @DataIn
101    public void setUsername(final String username) {
102        this.username = username;
103    }
104
105    @DataIn
106    public void setPassword(final String password) {
107        this.password = password;
108    }
109
110    @DataOut
111    public io.vertx.mqtt.MqttClient getClient() {
112        return this.client;
113    }
114}
