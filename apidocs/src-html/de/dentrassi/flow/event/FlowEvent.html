011package de.dentrassi.flow.event;
012
013import java.util.List;
014import java.util.Map;
015import java.util.stream.Collectors;
016
017import de.dentrassi.flow.ComponentInstance;
018import de.dentrassi.flow.Port;
019import de.dentrassi.flow.PortType;
020
021public interface FlowEvent {
022
023    public static FlowEvent addComponent(final String id, final String type) {
024        return new AddComponent(id, type);
025    }
026
027    public static FlowEvent addDataConnection(final String id, final Port out, final Port in) {
028        return new AddDataConnection(id, out, in);
029    }
030
031    public static FlowEvent addTriggerConnection(final String id, final Port out, final Port in) {
032        return new AddTriggerConnection(id, out, in);
033    }
034
035    public static FlowEvent addPort(final Port port, final PortType type) {
036        return new AddPort(port, type);
037    }
038
039    public static List<FlowEvent> addPorts(final ComponentInstance component,
040            final Map<String, PortType> reportedPorts) {
041
042        return reportedPorts
043                .entrySet()
044                .stream()
045                .map(entry -> addPort(component.port(entry.getKey()), entry.getValue()))
046                .collect(Collectors.toList());
047
048    }
049
050}
