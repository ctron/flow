011package de.dentrassi.flow.event;
012
013public class AddComponent implements FlowEvent {
014
015    private final String id;
016    private final String type;
017
018    AddComponent(final String id, final String type) {
019        this.id = id;
020        this.type = type;
021    }
022
023    public String getId() {
024        return this.id;
025    }
026
027    public String getType() {
028        return this.type;
029    }
030
031    @Override
032    public String toString() {
033        return new StringBuilder("[ADD_COMPONENT - ")
034                .append("ID: ").append(this.id).append(", ")
035                .append("Type: ").append(this.type)
036                .append("]")
037                .toString();
038    }
039
040}
