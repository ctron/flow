011package de.dentrassi.flow;
012
013import java.util.LinkedList;
014import java.util.Objects;
015import java.util.concurrent.atomic.AtomicLong;
016import java.util.function.Consumer;
017
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021import de.dentrassi.flow.internal.FlowExecutor;
022import de.dentrassi.flow.internal.FlowRunner;
023import de.dentrassi.flow.spi.type.ComponentFactory;
024
025public class Flow implements AutoCloseable {
026
027    private static class FlowExecutorImpl implements FlowExecutor {
028
029        private static final AtomicLong THREAD_COUNTER = new AtomicLong();
030
031        private boolean running;
032
033        private final LinkedList<Runnable> contextTasks = new LinkedList<>();
034
035        private final Thread thread;
036
037        public FlowExecutorImpl() {
038            this.thread = new Thread(this::contextRunner);
039            this.thread.setName("flow-" + THREAD_COUNTER.getAndIncrement());
040        }
041
042        public void start() {
043            this.thread.start();
044        }
045
046        public void stop() throws InterruptedException {
047            synchronized (this) {
048                this.running = false;
049                notifyAll();
050            }
051
052            this.thread.join();
053        }
054
055        @Override
056        public void execute(final Runnable command) {
057            Objects.requireNonNull(command);
058
059            synchronized (this) {
060                this.contextTasks.add(command);
061                notifyAll();
062            }
063        }
064
065        @Override
066        public boolean isExecutorThread() {
067            return Thread.currentThread().equals(this.thread);
068        }
069
070        protected synchronized void contextRunner() {
071            while (this.running) {
072
073                // process first, we might already have tasks when we got created
074
075                /*
076                 * Do this until we have an empty task list. It may be that executing tasks
077                 * actually creates new tasks in the context thread. Which we need to process as
078                 * well.
079                 */
080
081                while (!this.contextTasks.isEmpty()) {
082                    processContextTasks();
083                }
084
085                // now wait
086
087                try {
088                    wait();
089                } catch (final InterruptedException e) {
090                }
091            }
092        }
093
094        private void processContextTasks() {
095
096            // make a copy
097
098            final Runnable[] tasks = this.contextTasks.toArray(new Runnable[this.contextTasks.size()]);
099            this.contextTasks.clear();
100
101            // run tasks
102
103            for (final Runnable runnable : tasks) {
104                try {
105                    runnable.run();
106                } catch (final Exception e) {
107                    logger.info("Context task failed", e);
108                }
109            }
110        }
111
112    }
113
114    private static final Logger logger = LoggerFactory.getLogger(Flow.class);
115
116    private final FlowRunner runner;
117    private final FlowExecutorImpl executor;
118
119    public Flow(final ComponentFactory componentFactory) {
120        this.executor = new FlowExecutorImpl();
121
122        this.runner = new FlowRunner(this.executor, componentFactory);
123
124        this.executor.start();
125    }
126
127    public void start() {
128        this.executor.execute(this.runner::start);
129    }
130
131    public void stop() {
132        this.executor.execute(this.runner::stop);
133    }
134
135    public void modify(final Consumer<FlowContext> consumer) {
136        Objects.requireNonNull(consumer);
137
138        this.executor.execute(() -> {
139            consumer.accept(this.runner);
140        });
141    }
142
143    @Override
144    public void close() throws Exception {
145        this.executor.stop();
146    }
147}
